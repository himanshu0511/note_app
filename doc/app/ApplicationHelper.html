def error_tag(error, html)
  unless error.nil?
    raw("<div class='field_with_errors'>#{html}</div>")
  else
    raw(html)
  end
end

def parent_layout(layout)
  @view_flow.set(:layout, self.output_buffer)
  self.output_buffer = render(:file => "layouts/#{layout}")
end
